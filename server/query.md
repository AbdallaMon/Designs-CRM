CREATE TABLE MeetingReminder (
id INT AUTO_INCREMENT PRIMARY KEY,
clientLeadId INT NOT NULL,
time DATETIME NOT NULL,
status ENUM('IN_PROGRESS', 'DONE', 'MISSED') DEFAULT 'IN_PROGRESS',
meetingResult TEXT,
reminderReason TEXT,
userId INT NOT NULL,
createdAt DATETIME DEFAULT CURRENT_TIMESTAMP,
updatedAt DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

INDEX idx_clientLeadId (clientLeadId),
INDEX idx_userId (userId),

CONSTRAINT fk_meeting_clientLead FOREIGN KEY (clientLeadId) REFERENCES ClientLead(id) ON DELETE CASCADE,
CONSTRAINT fk_meeting_user FOREIGN KEY (userId) REFERENCES User(id) ON DELETE CASCADE
);
