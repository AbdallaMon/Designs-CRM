CREATE TABLE `Commission` (
`id` INT AUTO_INCREMENT PRIMARY KEY,
`userId` INT NOT NULL,
`leadId` INT NOT NULL,
`amount` DECIMAL(10,2) NOT NULL,
`amountPaid` DECIMAL(10,2) NOT NULL DEFAULT 0,
`createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
`isCleared` BOOLEAN NOT NULL DEFAULT FALSE,

FOREIGN KEY (`userId`) REFERENCES `User`(`id`) ON DELETE CASCADE,
FOREIGN KEY (`leadId`) REFERENCES `ClientLead`(`id`) ON DELETE CASCADE,
INDEX (`userId`),
INDEX (`leadId`)
);

ALTER TABLE `Note`
ADD COLUMN `commissionId` INT NULL,
ADD CONSTRAINT `fk_note_commission` FOREIGN KEY (`commissionId`) REFERENCES `Commission`(`id`) ON DELETE SET NULL,
ADD INDEX (`commissionId`);

ALTER TABLE Commission
ADD COLUMN commissionReason VARCHAR(255) NOT NULL DEFAULT 'Finalized Lead Commission';
